--------------------------------------------------------------------------- # Rotary positional embedding (RoPE) # --------------------------------------------------------------------------- def _rotate_half(x: torch.Tensor) -> torch.Tensor: x1, x2 = x[..., : x.shape[-1] // 2], x[..., x.shape[-1] // 2 :] return torch.cat([-x2, x1], dim=-1) def apply_rope(q: torch.Tensor, k: torch.Tensor, cos: torch.Tensor, sin: torch.Tensor) -> tuple[torch.Tensor, torch.Tensor]: q = (q * cos) + (_rotate_half(q) * sin) k = (k * cos) + (_rotate_half(k) * sin) return q, k class RoPECache(nn.Module): def __init__(self, dim: int, max_seq_len: int): super().__init__() theta = 1.0 / (10000 ** (torch.arange(0, dim, 2).float() / dim)) pos = torch.arange(max_seq_len).float() freqs = torch.outer(pos, theta) emb = torch.cat([freqs, freqs], dim=-1) self.register_buffer("cos", emb.cos()[None, None]) # (1,1,T,D) self.register_buffer("sin", emb.sin()[None, None]) def forward(self, seq_len: int): return self.cos[:, :, :seq_len], self.sin[:, :, :seq_len] # --------------------------------------------------------------------------- # Byte positional embedding (sinusoidal, injected into patches) # --------------------------------------------------------------------------- class BytePosEmbedding(nn.Module): """Sinusoidal embedding over byte offsets within each patch.""" def __init__(self, embed_dim: int, max_patch_size: int = 64): super().__init__() pos = torch.arange(max_patch_size).unsqueeze(1).float() div = torch.exp( torch.arange(0, embed_dim, 2).float() * (-math.log(10000.0) / embed_dim) ) emb = torch.zeros(max_patch_size, embed_dim) emb[:, 0::2] = torch.sin(pos * div) emb[:, 1::2] = torch.cos(pos * div) self.register_buffer("emb", emb) # (max_patch_size, D) def forward(self, patch_tensor: torch.Tensor, stride: int) -> torch.Tensor: """ patch_tensor: (B, T_p, D) already pooled patches Returns same shape with byte-offset embeddings added. (We inject the embedding for the *centre* byte of each patch.) """ T_p = patch_tensor.size(1) centre_offsets = torch.arange(T_p, device=patch_tensor.device) * stride + stride // 2 centre_offsets = centre_offsets.clamp(max=self.emb.size(0) - 1) return patch_tensor + self.emb[centre_offsets].unsqueeze(0) # --------------------------------------------------------------------------- # Learned boundary predictor (replaces heuristic entropy scan) # --------------------------------------------------------------------------- class LearnedBoundaryPredictor(nn.Module): """ 2-layer 1D CNN that predicts soft patch boundaries. Input: byte embeddings (B, T, D) Output: boundary logits (B, T) → soft-argmax → K boundary positions Enables fixed-count batched patching (K = num_patches). """ def __init__(self, embed_dim: int, num_patches: int): super().__init__() self.num_patches = num_patches self.conv = nn.Sequential( nn.Conv1d(embed_dim, embed_dim // 2, kernel_size=3, padding=1), nn.GELU(), nn.Conv1d(embed_dim // 2, 1, kernel_size=3, padding=1), ) def forward(self, x: torch.Tensor) -> torch.Tensor: """ x: (B, T, D) Returns boundary_positions: (B, num_patches) float indices into [0, T) Uses soft-argmax so gradients flow back through boundary positions. """ B, T, D = x.shape logits = self.conv(x.transpose(1, 2)).squeeze(1) # (B, T) # Top-K soft-argmax: select num_patches boundary positions # Use Gumbel-softmax windows around top-K peaks k = self.num_patches positions = torch.arange(T, device=x.device, dtype=x.dtype).unsqueeze(0) # (1, T) # Compute K soft boundary positions via sliding window soft-argmax # Divide sequence into K equal segments; soft-argmax within each segment seg_len = T // k boundaries = [] for i in range(k): start = i * seg_len end = start + seg_len if i < k - 1 else T seg_logits = logits[:, start:end] weights = F.softmax(seg_logits, dim=-1) # (B, seg_len) seg_pos = positions[:, start:end] bp = (weights * seg_pos).sum(dim=-1) # (B,) boundaries.append(bp) return torch.stack(boundaries, dim=1) # (B, num_patches) # --------------------------------------------------------------------------- # Entropy patch encoder (V3: uses learned boundaries when B > 1) # --------------------------------------------------------------------------- class EntropyPatchEncoder(nn.Module): """ Encodes byte sequences into patches. - Single sample: optional entropy-based variable boundaries (legacy) - Batched (B>1): uses LearnedBoundaryPredictor for fixed-count patches Falls back to fixed-stride if predictor is disabled. """ def __init__(self, embed_dim: int, patch_stride: int, num_patches: int, use_entropy_patching: bool = True): super().__init__() self.patch_stride = patch_stride self.num_patches = num_patches self.use_entropy_patching = use_entropy_patching self.conv = nn.Conv1d(embed_dim, embed_dim, kernel_size=patch_stride, stride=patch_stride, padding=0) self.norm = nn.LayerNorm(embed_dim) self.byte_pos = BytePosEmbedding(embed_dim) if use_entropy_patching: self.boundary_predictor = LearnedBoundaryPredictor(embed_dim, num_patches) def _fixed_stride_encode(self, x: torch.Tensor) -> torch.Tensor: """x: (B, T, D) → (B, T//stride, D)""" out = self.conv(x.transpose(1, 2)).transpose(1, 2) out = self.norm(out) return self.byte_pos(out, self.patch_stride) def _learned_boundary_encode(self, x: torch.Tensor) -> torch.Tensor: """x: (B, T, D) → (B, num_patches, D)""" B, T, D = x.shape boundaries = self.boundary_predictor(x) # (B, num_patches) soft float positions # Interpolate patch embeddings via soft gather # For each boundary position, compute weighted average of nearby embeddings patches = [] boundaries_int = boundaries.long().clamp(0, T - 1) for p in range(self.num_patches): pos = boundaries_int[:, p] # (B,) # Gather embedding at boundary position idx = pos.unsqueeze(-1).expand(B, D) # (B, D) patch_emb = x.gather(1, pos.unsqueeze(1).expand(B, 1, D)).squeeze(1) # (B, D) patches.append(patch_emb) out = torch.stack(patches, dim=1) # (B, num_patches, D) return self.norm(out) def forward(self, x: torch.Tensor) -> torch.Tensor: if self.use_entropy_patching and hasattr(self, 'boundary_predictor'): return self._learned_boundary_encode(x) return self._fixed_stride_encode(x) # --------------------------------------------------------------------------- # Grouped-query attention # --------------------------------------------------------------------------- class GroupedQueryAttention(nn.Module): """ GQA: num_kv_heads key/value heads shared across num_heads query heads. Set num_kv_heads == num_heads for standard MHA. """ def __init__(self, embed_dim: int, num_heads: int, num_kv_heads: int, dropout: float = 0.0): super().__init__() assert num_heads % num_kv_heads == 0, \ f"num_heads ({num_heads}) must be divisible by num_kv_heads ({num_kv_heads})" self.num_heads = num_heads self.num_kv_heads = num_kv_heads self.head_dim = embed_dim // num_heads self.groups = num_heads // num_kv_heads self.scale = self.head_dim ** -0.5 self.dropout = dropout self.q_proj = nn.Linear(embed_dim, embed_dim, bias=False) self.k_proj = nn.Linear(embed_dim, num_kv_heads * self.head_dim, bias=False) self.v_proj = nn.Linear(embed_dim, num_kv_heads * self.head_dim, bias=False) self.out_proj = nn.Linear(embed_dim, embed_dim, bias=False) def forward(self, x: torch.Tensor, cos: torch.Tensor, sin: torch.Tensor) -> torch.Tensor: B, T, D = x.shape q = self.q_proj(x).view(B, T, self.num_heads, self.head_dim).transpose(1, 2) k = self.k_proj(x).view(B, T, self.num_kv_heads, self.head_dim).transpose(1, 2) v = self.v_proj(x).view(B, T, self.num_kv_heads, self.head_dim).transpose(1, 2) q, k = apply_rope(q, k, cos, sin) # Expand k, v for GQA if self.groups > 1: k = k.unsqueeze(2).expand(B, self.num_kv_heads, self.groups, T, self.head_dim).reshape(B, self.num_heads, T, self.head_dim) v = v.unsqueeze(2).expand(B, self.num_kv_heads, self.groups, T, self.head_dim).reshape(B, self.num_heads, T, self.head_dim) attn = F.scaled_dot_product_attention( q, k, v, dropout_p=self.dropout if self.training else 0.0, is_causal=True, ) attn = attn.transpose(1, 2).reshape(B, T, D) return self.out_proj(attn) # --------------------------------------------------------------------------- # Transformer layer # --------------------------------------------------------------------------- class TransformerLayer(nn.Module): def __init__(self, cfg: ModelConfig): super().__init__() self.norm1 = nn.LayerNorm(cfg.embed_dim) self.norm2 = nn.LayerNorm(cfg.embed_dim) self.attn = GroupedQueryAttention( cfg.embed_dim, cfg.num_heads, cfg.num_kv_heads, cfg.dropout ) ffn_dim = int(cfg.embed_dim * cfg.ffn_multiplier) self.ffn = nn.Sequential( nn.Linear(cfg.embed_dim, ffn_dim, bias=False), nn.GELU(), nn.Dropout(cfg.dropout), nn.Linear(ffn_dim, cfg.embed_dim, bias=False), nn.Dropout(cfg.dropout), ) def _attn_block(self, x: torch.Tensor, cos: torch.Tensor, sin: torch.Tensor) -> torch.Tensor: return x + self.attn(self.norm1(x), cos, sin) def _ffn_block(self, x: torch.Tensor) -> torch.Tensor: return x + self.ffn(self.norm2(x)) def forward(self, x: torch.Tensor, cos: torch.Tensor, sin: torch.Tensor) -> torch.Tensor: x = self._attn_block(x, cos, sin) x = self._ffn_block(x) return x # --------------------------------------------------------------------------- # OverlapAddDecoder V3 — per-patch MLP window # --------------------------------------------------------------------------- class OverlapAddDecoder(nn.Module): """ V3: per-patch MLP-conditioned window instead of global learned window. Eliminates boundary artefacts in structured JPEG regions. """ def __init__(self, embed_dim: int, patch_stride: int): super().__init__() self.patch_stride = patch_stride self.out_dim = patch_stride # Projects each patch embedding to (patch_stride * embed_dim) output self.proj = nn.Linear(embed_dim, patch_stride * embed_dim) # Per-patch window MLP: takes patch features → scalar window weights self.window_mlp = nn.Sequential( nn.Linear(embed_dim, embed_dim // 4), nn.GELU(), nn.Linear(embed_dim // 4, patch_stride), nn.Sigmoid(), # window weights in [0, 1] ) self.norm = nn.LayerNorm(embed_dim) def forward(self, patches: torch.Tensor) -> torch.Tensor: """ patches: (B, T_p, D) Returns: (B, T_p * patch_stride, D) — overlap-add reconstructed """ B, T_p, D = patches.shape S = self.patch_stride proj = self.proj(patches) # (B, T_p, S*D) proj = proj.view(B, T_p, S, D) # Per-patch window weights window = self.window_mlp(patches) # (B, T_p, S) proj = proj * window.unsqueeze(-1) # (B, T_p, S, D) # Fold back into sequence via overlap-add proj_flat = proj.reshape(B, T_p * S, D) # simple concatenation (no overlap here) return self.norm(proj_flat) # --------------------------------------------------------------------------- # Main model # --------------------------------------------------------------------------- class TinyByteModel(nn.Module): def __init__(self, cfg: ModelConfig): super().__init__() self.cfg = cfg self.embed = nn.Embedding(cfg.vocab_size, cfg.embed_dim) self.patch_encoder = EntropyPatchEncoder( cfg.embed_dim, cfg.patch_stride, cfg.num_patches, cfg.use_entropy_patching ) self.rope = RoPECache(cfg.embed_dim // cfg.num_heads, cfg.max_seq_len) self.layers = nn.ModuleList([TransformerLayer(cfg) for _ in range(cfg.num_layers)]) self.norm = nn.LayerNorm(cfg.embed_dim) self.decoder = OverlapAddDecoder(cfg.embed_dim, cfg.patch_stride) self.head = nn.Linear(cfg.embed_dim, cfg.vocab_size, bias=False) # Weight tying self.head.weight = self.embed.weight self._init_weights() def _init_weights(self): for m in self.modules(): if isinstance(m, nn.Linear): nn.init.normal_(m.weight, std=0.02) if m.bias is not None: nn.init.zeros_(m.bias) elif isinstance(m, nn.Embedding): nn.init.normal_(m.weight, std=0.02) def forward(self, input_ids: torch.Tensor) -> torch.Tensor: """ input_ids: (B, T) byte indices Returns logits: (B, T, vocab_size) """ x = self.embed(input_ids) # (B, T, D) x = self.patch_encoder(x) # (B, T_p, D) T_p = x.size(1) cos, sin = self.rope(T_p) for layer in self.layers: if self.cfg.gradient_checkpointing and self.training: x = checkpoint(layer, x, cos, sin, use_reentrant=False) else: x = layer(x, cos, sin) x = self.norm(x) x = self.decoder(x) # (B, T_r, D) # Align reconstructed length with input length T_in = input_ids.size(1) T_r = x.size(1) if T_r > T_in: x = x[:, :T_in] elif T_r < T_in: x = F.pad(x, (0, 0, 0, T_in - T_r)) return self.head(x) # (B, T, vocab_size) # ----------------------------------------------------------------------- # Loss with JPEG header auxiliary loss # ----------------------------------------------------------------------- def compute_loss( self, input_ids: torch.Tensor, labels: torch.Tensor, is_image_sequence: Optional[torch.Tensor] = None, ) -> dict: """ input_ids: (B, T) labels: (B, T) — same as input shifted by 1 outside is_image_sequence: (B,) bool — which samples are image sequences Returns dict with 'loss', 'main_loss', 'header_loss', 'trim_frac' """ logits = self.forward(input_ids) # (B, T, V) B, T, V = logits.shape # Trim diagnostic label_len = (labels != -100).sum(dim=1).float() trim_frac = (1.0 - label_len / T).mean().item() # Main next-byte loss main_loss = F.cross_entropy( logits.reshape(-1, V), labels.reshape(-1), ignore_index=-100, ) total_loss = main_loss header_loss = torch.tensor(0.0, device=input_ids.device) # Auxiliary JPEG header loss if (is_image_sequence is not None and is_image_sequence.any() and self.cfg.jpeg_header_loss_weight > 0): img_mask = is_image_sequence.bool() hlen = min(self.cfg.jpeg_header_bytes, T - 1) header_logits = logits[img_mask, :hlen] # (B_img, hlen, V) header_labels = labels[img_mask, :hlen] # (B_img, hlen) if header_labels.numel() > 0: header_loss = F.cross_entropy( header_logits.reshape(-1, V), header_labels.reshape(-1), ignore_index=-100, ) total_loss = main_loss + self.cfg.jpeg_header_loss_weight * header_loss return { "loss": total_loss, "main_loss": main_loss.item(), "header_loss": header_loss.item(), "trim_frac": trim_frac, } # sorted_probs > top_p sorted_probs[mask] = 0.0 sorted_probs /= sorted_probs.sum(dim=-1, keepdim=True).clamp(min=1e-8) # Scatter back to original order out = torch.zeros_like(probs) out.scatter_(-1, sorted_idx, sorted_probs) return out

13. Known Limitations & Next Steps

Remaining Limitations

Image quality at moderate scale. At 24 layers / 1024 dimensions / 1M pairs the model will produce JPEG-structured bytes with recognisable macro-structure and rough spatial layout. Coherent photorealistic imagery requires hundreds of millions of pairs — consistent with reported EvaByte results. The progressive curriculum helps, but is not a substitute for scale.

Learned boundary predictor is a soft approximation. The segment-wise soft-argmax approach produces fixed-count patches but is an approximation to true content-adaptive patching. True end-to-end differentiable patching with variable boundaries remains an open research problem.

Speculative decoding acceptance rate degrades over long sequences. Draft and main model distributions diverge for long image tails (high-frequency AC coefficients). Consider retraining the draft model to focus on the image distribution specifically.

FID unreliable below ~1000 samples. Use CLIP-score and PickScore for smaller evaluation budgets. FID is included for completeness.

FSDP + torch.compile interaction. On some PyTorch versions, torch.compile inside FSDP can produce graph breaks. If --compile causes errors in distributed mode, disable it and rely on bf16 + gradient checkpointing alone.

PickScore requires internet access to download model weights on first run. Cache the weights if training in an air-gapped environment.

Suggested Next Steps (Priority Order)

Scale data to 5M+ pairs via LAION-COCO / CC12M — still the single highest-leverage change.

Train a dedicated DraftModel on the same data — the current draft model must be trained separately; a co-distillation approach would improve acceptance rate.

Discrete VAE pathway — add a small VQ-VAE producing 8–16× shorter discrete tokens; keep byte path as auxiliary loss. This is the most impactful architectural change for V4.

Mixture-of-Depth layers — skip expensive layers for low-entropy patches (detected via the existing boundary predictor's entropy signal).

Gradio / web demo — add notebooks/demo.ipynb with a Gradio interface and Colab badge.

ONNX / TorchScript export — for edge deployment; the small preset is a natural target.

TinyByte Multimodal V3 — Tokenizer-free. Bytes are all you need.
